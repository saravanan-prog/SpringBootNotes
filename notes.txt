 @PutMapping("/updateStaff/{id}")
    public List<StaffModel> updateStaff(
            @PathVariable long id,
            @RequestBody StaffModel updateStaffPayload
    ){

        return this.serviceobj.updateStaffDetails(updateStaffPayload,id);
    }



public List<StaffModel> updateStaffDetails(StaffModel staffPayload,long id){

         staff.stream()
                .filter( x-> x.getStaffId() == id )
                .peek(
                        item -> {
                            item.setStaffName(staffPayload.getStaffName());
                            item.setStaffAddress(staffPayload.getStaffAddress());
                        }
                )
                .collect(Collectors.toList());



        return staff;
    }
